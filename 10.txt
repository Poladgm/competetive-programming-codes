#include <cstdlib>
#include <iostream>
#include <math.h>
using namespace std;

double N, n;
int k;
double d, sum;
int main(int argc, char *argv[])
{
   cin>>N;
   n = N;
   while( sum <= 0.5 )
   {
   d = 1 / N;
   sum = sum + d;
   N = N - 1;
   k ++;
}
   cout<<n - k + 1<<endl;  
   
   return EXIT_SUCCESS;
}
