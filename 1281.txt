long long a[10000000];
long mod(long x,long  m) {
  long long r = x%m;
  return r<0 ? r+m : r;
}
long long name(long long n){
  if(a[n]) return a[n];
  a[(n-1)]=name((n-1));
  a[(n-4)]=name((n-4));
  return 2*a[(n-1)]-name((n-4));
}
