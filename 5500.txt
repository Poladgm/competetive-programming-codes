#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
    double a,b,x1,Y1,x2,y2,x3,y3,s1,s2,s3,s;
    scanf("%lf%lf%lf%lf%lf%lf%lf%lf",&x1,&Y1,&x2,&y2,&x3,&y3,&a,&b);

    if((a==x1 && b==Y1) || (a==x2 && b==y2) || (a==x3 && b==y3))
    {
        printf("Vertex\n");
        return 0;
    }

    s=abs(x1*y2+x2*y3+x3*Y1-x2*Y1-x3*y2-x1*y3);
    s1=abs(x1*b+a*y2+x2*Y1-a*Y1-b*x2-x1*y2);
    s2=abs(a*y2+x2*y3+x3*b-x2*b-x3*y2-a*y3);
    s3=abs(x1*b+a*y3+x3*Y1-a*Y1-x3*b-y3*x1);

    if((s1==0 || s2==0 || s3==0) && s==s1+s2+s3)
    printf("Edge\n");
    else if(s==s1+s2+s3)
    printf("In\n");
    else
    printf("Out\n");

    return 0;
}
