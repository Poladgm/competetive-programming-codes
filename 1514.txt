unsigned long long ust(unsigned long long a,unsigned long long b)
{
   if(b==1)
       return a%m;
   if(b%2==0)
       return ust(((a%m)*(a%m))%m,b/2)%m;
   else
       return ((a%m)*ust(((a%m)*(a%m))%m,b/2))%m;        
}

int ust2(int a,int b)
{
   int s=1;
   for(int i=0;i<b;i++)
       s*=a;
   return s;    
}
