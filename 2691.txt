long long cem,i,n,m,j,k,eded;

int main(int argc, char *argv[])
{
    scanf("%lld",&n);

        cem=0;

        scanf("%lld",&m);

            scanf("%lld",&eded);

        ortaqiymet[i]=(double)(cem/m);

    }
