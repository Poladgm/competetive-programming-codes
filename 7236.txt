#include <iostream>
#include <set>
#include <vector>
#include <stdio.h>

using namespace std;

vector <int> g[100005];
vector <int> temp;
int used[100005];
int x,y,k,n,m,ans;

void dfs(int x)
{
   used[x] = 1;

   for(int i=0; i<g[x].size(); i++)
   {
       if(used[g[x][i]] == 0)
       {
           dfs(g[x][i]);
       }
   }
}

int main()
{
   ios_base::sync_with_stdio(false);
   cin>>n>>m;
   for(int i = 1; i <= m; i++)
   {
       cin>>x>>y;
       g[x].push_back(y);
       g[y].push_back(x);
   }

   for(int i = 1; i<=n; i++)
   {
       if(used[i] == 0)
       {
           ans++;
           dfs(i);
       }
   }


   cout<<ans<<endl;


   return 0;
}
